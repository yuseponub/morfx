    ```typescript
    import { META_BASE_URL } from './constants'
    export async function exchangeCodeForBisuat(code: string): Promise<string> {
      const url = `${META_BASE_URL}/oauth/access_token`
        + `?client_id=${process.env.META_APP_ID}`
        + `&client_secret=${process.env.META_APP_SECRET}`
        + `&code=${encodeURIComponent(code)}`
      const res = await fetch(url)            // NO Authorization header (Pitfall 6)
      const data = await res.json()
      if (!res.ok || !data.access_token) {
        throw new Error(`exchange failed: ${JSON.stringify(data)}`)
      }
      return data.access_token as string      // BISUAT — no expiry, can be invalidated
    }
    ```

    Auth gate — COPY from shopify-oauth.ts:70-93 (V4 access control). Derive `workspaceId` from the `morfx_workspace` cookie, NEVER from `input` (security threat "Unauthorized workspace connecting a number" — RESEARCH:600):
    ```typescript
    'use server'
    import { cookies } from 'next/headers'
    import { createClient } from '@/lib/supabase/server'
    import { encryptToken } from '@/lib/meta/token'
    import { exchangeCodeForBisuat, subscribeWaba } from '@/lib/meta/embedded-signup'
    import { upsertMetaAccount } from '@/lib/domain/meta-accounts'

    export async function connectWhatsAppNumber(
      input: { code: string; wabaId: string; phoneNumberId: string }
    ): Promise<{ success: true } | { success: false; error: string }> {
      const supabase = await createClient()
      const { data: { user } } = await supabase.auth.getUser()
      if (!user) return { success: false, error: 'No autenticado' }
      const cookieStore = await cookies()
      const workspaceId = cookieStore.get('morfx_workspace')?.value
      if (!workspaceId) return { success: false, error: 'No hay workspace seleccionado' }
      const { data: member } = await supabase
        .from('workspace_members').select('role')
        .eq('workspace_id', workspaceId).eq('user_id', user.id).single()
      if (!member || member.role !== 'owner') {
        return { success: false, error: 'Solo el Owner puede conectar WhatsApp' }
      }

      // Input validation (V5)
      if (!input.code || !input.wabaId || !input.phoneNumberId) {
        return { success: false, error: 'Datos de conexión incompletos' }
      }

      try {
        const bisuat = await exchangeCodeForBisuat(input.code)        // server-side, secret stays server (SIGNUP-02)
        const accessTokenEncrypted = encryptToken(bisuat)            // AES-256-GCM
        const result = await upsertMetaAccount({                     // Regla 3 — domain write
          workspaceId, channel: 'whatsapp',
          wabaId: input.wabaId, phoneNumberId: input.phoneNumberId,
          accessTokenEncrypted, isActive: true,
        })
        if (!result.success) return { success: false, error: result.error }
        await subscribeWaba(bisuat, input.wabaId)                    // SIGNUP-03 auto-subscribe
        return { success: true }
      } catch (e) {
        console.error('[meta-onboarding] connect failed:', e)        // detail server-side only
        return { success: false, error: 'No se pudo conectar el número de WhatsApp. Intenta de nuevo.' }
      }
    }
    ```
