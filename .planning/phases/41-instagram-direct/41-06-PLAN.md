```tsx
'use client'
import { useTransition } from 'react'
import { Button } from '@/components/ui/button'
import { Instagram, Loader2 } from 'lucide-react'
import { toast } from 'sonner'
import { connectInstagramAccount } from '@/app/actions/meta-onboarding'

export function ConnectInstagram() {
  const [pending, startTransition] = useTransition()
  function handleConnect() {
    startTransition(async () => {
      const result = await connectInstagramAccount()
      if (result.success) {
        toast.success(result.igUsername ? `Instagram conectado: @${result.igUsername}` : 'Instagram conectado')
      } else {
        toast.error(result.error ?? 'No se pudo conectar Instagram')
      }
    })
  }
  return (
    <Button onClick={handleConnect} disabled={pending} className="w-full">
      {pending ? <Loader2 className="mr-2 h-4 w-4 animate-spin" /> : <Instagram className="mr-2 h-4 w-4" />}
      Conectar Instagram
    </Button>
  )
}
```

The browser never sees a token (the action resolves IG off the stored encrypted Page token server-side). Use the lucide `Instagram` icon + Spanish copy. Match the styling of the FB button (full-width, spinner while pending).
