    import createMiddleware from 'next-intl/middleware';
    import { NextRequest } from 'next/server';
    import { routing } from '@/i18n/routing';
    import { updateSession } from '@/lib/supabase/middleware';

    const intlMiddleware = createMiddleware(routing);

    const PUBLIC_MARKETING = [
      '/', '/en',
      '/privacy', '/en/privacy',
      '/terms', '/en/terms',
    ];

    export async function middleware(request: NextRequest) {
      const pathname = request.nextUrl.pathname;
      const isPublicMarketing = PUBLIC_MARKETING.includes(pathname);

      if (isPublicMarketing) {
        // next-intl handles locale detection/rewrites for marketing routes
        return intlMiddleware(request);
      }

      // All other routes go through Supabase session flow (auth + protected)
      return updateSession(request);
    }

    export const config = {
      // Skip static assets, API routes, and Next internals
      matcher: [
        '/((?!api|_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp|ico)$).*)',
      ],
    };

  </action>
