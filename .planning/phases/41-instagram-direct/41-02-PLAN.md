```typescript
import { metaRequest } from './api'

export type InstagramTag = 'HUMAN_AGENT'   // IG supports ONLY HUMAN_AGENT (same as FB)
interface InstagramSendResponse { message_id?: string; recipient_id?: string }

export async function sendInstagramText(accessToken: string, pageId: string, igsid: string, text: string, tag?: InstagramTag) {
  const body = tag
    ? { messaging_type: 'MESSAGE_TAG', tag: 'HUMAN_AGENT', recipient: { id: igsid }, message: { text } }
    : { messaging_type: 'RESPONSE',                          recipient: { id: igsid }, message: { text } }
  return metaRequest<InstagramSendResponse>(accessToken, `/${pageId}/messages`, { method: 'POST', body: JSON.stringify(body) })
}

export async function sendInstagramImage(accessToken: string, pageId: string, igsid: string, imageUrl: string, tag?: InstagramTag) {
  const body = {
    messaging_type: tag ? 'MESSAGE_TAG' : 'RESPONSE',
    ...(tag ? { tag: 'HUMAN_AGENT' } : {}),
    recipient: { id: igsid },
    message: { attachment: { type: 'image', payload: { url: imageUrl, is_reusable: true } } },
  }
  return metaRequest<InstagramSendResponse>(accessToken, `/${pageId}/messages`, { method: 'POST', body: JSON.stringify(body) })
}

export async function sendInstagramAttachment(accessToken: string, pageId: string, igsid: string, attachmentType: 'image'|'video'|'audio'|'file', mediaUrl: string, tag?: InstagramTag) {
  const body = {
    messaging_type: tag ? 'MESSAGE_TAG' : 'RESPONSE',
    ...(tag ? { tag: 'HUMAN_AGENT' } : {}),
    recipient: { id: igsid },
    message: { attachment: { type: attachmentType, payload: { url: mediaUrl, is_reusable: true } } },
  }
  return metaRequest<InstagramSendResponse>(accessToken, `/${pageId}/messages`, { method: 'POST', body: JSON.stringify(body) })
}

// IG direct edge — works with the Page token + instagram_basic + instagram_manage_messages (APPROVED).
export async function getInstagramUserName(accessToken: string, igsid: string): Promise<string | null> {
  try {
    const p = await metaRequest<{ name?: string; username?: string }>(accessToken, `/${igsid}?fields=name,username`)
    return p.name?.trim() || (p.username ? `@${p.username}` : null)
  } catch {
    return null
  }
}
```

import { sendInstagramText, sendInstagramImage, sendInstagramAttachment } from '../meta/instagram-api'
import type { ChannelSendResult } from './types'

export interface MetaPageCreds { accessToken: string; pageId: string }   // IG uses the Page token + Page id
interface SendResponse { message_id?: string }
function unwrap(r: SendResponse): ChannelSendResult { return { success: true, externalMessageId: r.message_id } }

export const metaInstagramSender = {
  async sendText(creds: MetaPageCreds, igsid: string, text: string, tag?: 'HUMAN_AGENT') {
    return unwrap(await sendInstagramText(creds.accessToken, creds.pageId, igsid, text, tag))
  },
  async sendImage(creds: MetaPageCreds, igsid: string, imageUrl: string, caption?: string, tag?: 'HUMAN_AGENT') {
    const r = await sendInstagramImage(creds.accessToken, creds.pageId, igsid, imageUrl, tag)
    if (caption) await sendInstagramText(creds.accessToken, creds.pageId, igsid, caption, tag)
    return unwrap(r)
  },
  async sendMedia(creds: MetaPageCreds, igsid: string, mediaType: 'image'|'video'|'audio'|'document', mediaUrl: string, caption?: string, tag?: 'HUMAN_AGENT') {
    if (mediaType === 'image') return this.sendImage(creds, igsid, mediaUrl, caption, tag)
    const attachmentType = mediaType === 'document' ? 'file' : mediaType
    const r = await sendInstagramAttachment(creds.accessToken, creds.pageId, igsid, attachmentType, mediaUrl, tag)
    if (caption) await sendInstagramText(creds.accessToken, creds.pageId, igsid, caption, tag)
    return unwrap(r)
  },
}
